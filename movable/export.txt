import html2canvas from 'html2canvas';
import jsPDF from 'jspdf';
import { PDFDocument, rgb, PDFString, PDFName } from 'pdf-lib';
import Papa from 'papaparse';
import { UploadedFile, Annotation, DocumentMetadata, Severity } from '../types';
import { SEVERITY_COLORS, SEVERITY_LABELS } from '../constants';

const hexToRgb = (hex: string) => {
  const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
  return result ? {
    r: parseInt(result[1], 16) / 255,
    g: parseInt(result[2], 16) / 255,
    b: parseInt(result[3], 16) / 255
  } : { r: 1, g: 0, b: 0 };
};

export const exportDocument = async (
  element: HTMLElement, 
  fileInfo: UploadedFile | null,
  annotations: Annotation[] = [],
  metadata: DocumentMetadata
) => {
  if (!fileInfo) return;

  try {
    const baseFileName = fileInfo.name ? fileInfo.name.split('.')[0] : 'reviewed_document';
    
    // 1. Export PDF
    if (fileInfo.type === 'pdf') {
       await exportPdfNative(fileInfo, annotations, `${baseFileName}_reviewed.pdf`);
    } else {
       if (!element) return;
       await exportImageAsPdf(element, annotations, `${baseFileName}_reviewed.pdf`);
    }

    // 2. Export CSV
    exportCsv(annotations, metadata, `${baseFileName}_comments.csv`);

  } catch (error) {
    console.error('Failed to export:', error);
    alert('Failed to generate export files. Please try again.');
  }
};

const exportPdfNative = async (fileInfo: UploadedFile, annotations: Annotation[], fileName: string) => {
  // Read original PDF
  const arrayBuffer = await fileInfo.fileObject.arrayBuffer();
  const pdfDoc = await PDFDocument.load(arrayBuffer);
  const pages = pdfDoc.getPages();

  // Draw annotations on corresponding pages
  annotations.forEach(ann => {
    // pageNumber is 1-indexed
    const pageIndex = (ann.pageNumber || 1) - 1;
    if (pageIndex >= 0 && pageIndex < pages.length) {
      const page = pages[pageIndex];
      const { width, height } = page.getSize();
      
      // Calculate coordinates (pdf-lib y is from bottom, our y is from top)
      const rectX = (ann.x / 100) * width;
      const rectW = (ann.width / 100) * width;
      const rectH = (ann.height / 100) * height;
      const rectY = height - ((ann.y / 100) * height) - rectH;

      const color = hexToRgb(SEVERITY_COLORS[ann.severity]);

      // Draw bounding box
      page.drawRectangle({
        x: rectX,
        y: rectY,
        width: rectW,
        height: rectH,
        borderColor: rgb(color.r, color.g, color.b),
        borderWidth: 2,
        color: rgb(color.r, color.g, color.b),
        opacity: 0.2, // semi-transparent fill
        borderOpacity: 1,
      });

      // We replace static text with a native PDF text annotation (sticky note)
      const annotObj = pdfDoc.context.obj({
        Type: 'Annot',
        Subtype: 'Text',
        Rect: [rectX, rectY, rectX + rectW, rectY + rectH],
        Contents: PDFString.of(ann.comment),
        T: PDFString.of(`${SEVERITY_LABELS[ann.severity]} Comment`),
        Open: false,
        Name: PDFName.of('Note')
      });
      const annotRef = pdfDoc.context.register(annotObj);
      page.node.addAnnot(annotRef);
    }
  });

  const pdfBytes = await pdfDoc.save();
  const blob = new Blob([pdfBytes as any], { type: 'application/pdf' });
  downloadBlob(blob, fileName);
};

const exportImageAsPdf = async (element: HTMLElement, annotations: Annotation[], fileName: string) => {
  const canvas = await html2canvas(element, {
    scale: 2,
    useCORS: true,
    logging: false,
    backgroundColor: '#ffffff'
  });

  const imgData = canvas.toDataURL('image/png');
  const width = canvas.width;
  const height = canvas.height;

  const pdf = new jsPDF({
    orientation: width > height ? 'l' : 'p',
    unit: 'px',
    format: [width, height]
  });

  pdf.addImage(imgData, 'PNG', 0, 0, width, height);

  // Add invisible interactive PDF annotations
  annotations.forEach(ann => {
    const rectX = (ann.x / 100) * width;
    const rectY = (ann.y / 100) * height;
    const rectW = (ann.width / 100) * width;
    const rectH = (ann.height / 100) * height;

    (pdf as any).createAnnotation({
      type: 'text',
      title: `${SEVERITY_LABELS[ann.severity]} Comment`,
      contents: ann.comment,
      bounds: { x: rectX, y: rectY, w: rectW, h: rectH },
      open: false
    });
  });

  pdf.save(fileName);
};

const exportCsv = (annotations: Annotation[], metadata: DocumentMetadata, fileName: string) => {
  const data = annotations.map((ann, i) => ({
    'Comment No.': i + 1,
    'Designer Name': metadata.designerName,
    'Job Number': metadata.jobNumber,
    'Delivery Type': metadata.deliveryType,
    'Page Number': ann.pageNumber || 1,
    'Severity': SEVERITY_LABELS[ann.severity],
    'Comment': ann.comment,
  }));

  const csv = Papa.unparse(data);
  const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
  downloadBlob(blob, fileName);
};

const downloadBlob = (blob: Blob, fileName: string) => {
  const link = document.createElement('a');
  if (link.download !== undefined) {
    const url = URL.createObjectURL(blob);
    link.setAttribute('href', url);
    link.setAttribute('download', fileName);
    link.style.visibility = 'hidden';
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
    URL.revokeObjectURL(url);
  }
};
