import React, { useState } from 'react';
import { ShapeType, UploadedFile, Annotation, DocumentMetadata } from '../types';
import { Square, Circle, Download, Upload, FileText, MessageSquare, Info, Pencil, Trash2, Check, X } from 'lucide-react';
import { SUPPORTED_IMAGE_TYPES, SUPPORTED_PDF_TYPES, SEVERITY_COLORS } from '../constants';

interface SidebarProps {
  file: UploadedFile | null;
  activeTool: ShapeType;
  setActiveTool: (tool: ShapeType) => void;
  onUpload: (e: React.ChangeEvent<HTMLInputElement>) => void;
  onDownload: (metadata: DocumentMetadata) => void;
  annotations: Annotation[];
  isExporting: boolean;
  onRemoveAnnotation: (id: string) => void;
  onUpdateAnnotation: (id: string, newComment: string) => void;
}

export const Sidebar: React.FC<SidebarProps> = ({ 
  file, 
  activeTool, 
  setActiveTool, 
  onUpload, 
  onDownload,
  annotations,
  isExporting,
  onRemoveAnnotation,
  onUpdateAnnotation
}) => {
  const [editingId, setEditingId] = useState<string | null>(null);
  const [tempComment, setTempComment] = useState('');
  
  const [designerName, setDesignerName] = useState('');
  const [jobNumber, setJobNumber] = useState('');
  const [deliveryType, setDeliveryType] = useState<'Complex' | 'Medium' | 'Low'>('Medium');

  const isFormValid = designerName.trim() !== '' && jobNumber.trim() !== '';

  const handleStartEdit = (id: string, currentComment: string) => {
    setEditingId(id);
    setTempComment(currentComment);
  };

  const handleSaveEdit = () => {
    if (editingId && tempComment.trim()) {
      onUpdateAnnotation(editingId, tempComment);
      setEditingId(null);
    }
  };

  const handleCancelEdit = () => {
    setEditingId(null);
    setTempComment('');
  };

  return (
    <div className="w-80 bg-white border-l border-gray-200 flex flex-col h-full shadow-lg z-20">
      
      {/* Header */}
      <div className="p-6 border-b border-gray-100 bg-gray-50">
        <h1 className="text-xl font-bold text-gray-800 flex items-center gap-2">
          <FileText className="text-brand-600" size={24} />
          DocuReview
        </h1>
        <p className="text-xs text-gray-500 mt-1">Review, Annotate, Export</p>
      </div>

      <div className="flex-1 overflow-y-auto p-6 space-y-8">
        
        {/* Upload Section */}
        <section>
          <h2 className="text-sm font-semibold text-gray-900 mb-3 uppercase tracking-wider">Document</h2>
          {file ? (
            <div className="bg-blue-50 border border-blue-100 rounded-lg p-3 flex items-start gap-3">
              <div className="p-2 bg-white rounded-md shadow-sm text-brand-600">
                <FileText size={20} />
              </div>
              <div className="flex-1 min-w-0">
                <p className="text-sm font-medium text-gray-900 truncate" title={file.name}>{file.name}</p>
                <p className="text-xs text-gray-500 mt-0.5 capitalize">{file.type} File</p>
                <label className="text-xs text-brand-600 hover:text-brand-700 cursor-pointer mt-2 inline-block font-medium">
                  Replace File
                  <input 
                    type="file" 
                    className="hidden" 
                    onChange={onUpload}
                    accept={[...SUPPORTED_IMAGE_TYPES, ...SUPPORTED_PDF_TYPES].join(',')} 
                  />
                </label>
              </div>
            </div>
          ) : (
            <label className="flex flex-col items-center justify-center w-full h-32 border-2 border-dashed border-gray-300 rounded-lg cursor-pointer hover:bg-gray-50 hover:border-brand-300 transition-colors group">
              <div className="flex flex-col items-center justify-center pt-5 pb-6">
                <Upload className="w-8 h-8 text-gray-400 group-hover:text-brand-500 mb-2" />
                <p className="text-sm text-gray-500 text-center px-2">
                  <span className="font-semibold text-brand-600">Click to upload</span>
                  <br/> PDF, JPG, PNG
                </p>
              </div>
              <input 
                type="file" 
                className="hidden" 
                onChange={onUpload} 
                accept={[...SUPPORTED_IMAGE_TYPES, ...SUPPORTED_PDF_TYPES].join(',')}
              />
            </label>
          )}
        </section>

        {/* Tools Section */}
        <section>
          <h2 className="text-sm font-semibold text-gray-900 mb-3 uppercase tracking-wider">Annotation Tools</h2>
          <div className="grid grid-cols-2 gap-3">
            <button
              onClick={() => setActiveTool(ShapeType.RECTANGLE)}
              className={`flex flex-col items-center justify-center p-3 rounded-lg border transition-all ${
                activeTool === ShapeType.RECTANGLE
                  ? 'bg-brand-50 border-brand-500 text-brand-700 ring-1 ring-brand-500'
                  : 'bg-white border-gray-200 text-gray-600 hover:border-brand-200 hover:bg-gray-50'
              }`}
            >
              <Square size={20} className="mb-1" />
              <span className="text-xs font-medium">Rectangle</span>
            </button>
            <button
              onClick={() => setActiveTool(ShapeType.CIRCLE)}
              className={`flex flex-col items-center justify-center p-3 rounded-lg border transition-all ${
                activeTool === ShapeType.CIRCLE
                  ? 'bg-brand-50 border-brand-500 text-brand-700 ring-1 ring-brand-500'
                  : 'bg-white border-gray-200 text-gray-600 hover:border-brand-200 hover:bg-gray-50'
              }`}
            >
              <Circle size={20} className="mb-1" />
              <span className="text-xs font-medium">Circle</span>
            </button>
          </div>
          <div className="mt-2 text-xs text-gray-500 flex items-start gap-1.5 bg-gray-50 p-2 rounded">
            <Info size={14} className="mt-0.5 flex-shrink-0" />
            <p>Select a tool, then click and drag on the document to highlight an area.</p>
          </div>
        </section>

        {/* Comments Summary */}
        <section className="flex-1 min-h-0 flex flex-col">
          <h2 className="text-sm font-semibold text-gray-900 mb-3 uppercase tracking-wider flex items-center justify-between">
            <span>Comments</span>
            <span className="bg-gray-100 text-gray-600 px-2 py-0.5 rounded-full text-xs">{annotations.length}</span>
          </h2>
          {annotations.length === 0 ? (
            <div className="flex-1 flex flex-col items-center justify-center text-center p-4 bg-gray-50 rounded-lg border border-gray-100 border-dashed h-32">
              <MessageSquare className="text-gray-300 mb-2" size={24} />
              <p className="text-sm text-gray-400">No comments yet</p>
            </div>
          ) : (
            <div className="space-y-3 max-h-60 overflow-y-auto pr-1">
              {annotations.map((ann, i) => (
                <div key={ann.id} className="p-3 bg-white border border-gray-200 rounded-lg shadow-sm text-sm hover:border-gray-300 transition-colors relative pl-4 group">
                  <div className="absolute left-0 top-0 bottom-0 w-1 bg-brand-500 rounded-l-lg"></div>
                  
                  {/* Comment Header with Index & Actions */}
                  <div className="flex justify-between items-center mb-1.5">
                    <div className="flex items-center gap-2">
                      <span className="inline-flex items-center justify-center w-5 h-5 bg-brand-100 text-brand-700 text-[10px] font-bold rounded-full">
                        {i + 1}
                      </span>
                      <span 
                        className="w-2.5 h-2.5 rounded-full shadow-sm"
                        style={{ backgroundColor: SEVERITY_COLORS[ann.severity] }}
                        title={`Severity: ${ann.severity}`}
                      />
                    </div>
                    
                    {/* Action Buttons (Visible on Hover) */}
                    {editingId !== ann.id && (
                      <div className="flex gap-1 opacity-100 sm:opacity-0 sm:group-hover:opacity-100 transition-opacity">
                        <button 
                          onClick={() => handleStartEdit(ann.id, ann.comment)}
                          className="p-1 text-gray-400 hover:text-brand-600 hover:bg-brand-50 rounded transition-colors"
                          title="Edit Comment"
                        >
                          <Pencil size={12} />
                        </button>
                        <button 
                          onClick={() => onRemoveAnnotation(ann.id)}
                          className="p-1 text-gray-400 hover:text-red-500 hover:bg-red-50 rounded transition-colors"
                          title="Delete Comment"
                        >
                          <Trash2 size={12} />
                        </button>
                      </div>
                    )}
                  </div>

                  {/* Comment Content or Edit Form */}
                  {editingId === ann.id ? (
                    <div className="space-y-2 animate-in fade-in duration-200">
                      <textarea
                        value={tempComment}
                        onChange={(e) => setTempComment(e.target.value)}
                        className="w-full p-2 border border-gray-300 rounded text-xs focus:ring-1 focus:ring-brand-500 focus:outline-none resize-none"
                        rows={3}
                        autoFocus
                        placeholder="Update your comment..."
                        onKeyDown={(e) => {
                          if (e.key === 'Enter' && !e.shiftKey) {
                            e.preventDefault();
                            handleSaveEdit();
                          }
                        }}
                      />
                      <div className="flex justify-end gap-2">
                        <button 
                          onClick={handleCancelEdit} 
                          className="p-1.5 text-gray-500 hover:bg-gray-100 rounded-md transition-colors"
                          title="Cancel"
                        >
                          <X size={14} />
                        </button>
                        <button 
                          onClick={handleSaveEdit} 
                          disabled={!tempComment.trim()}
                          className="p-1.5 text-white bg-brand-600 hover:bg-brand-700 rounded-md disabled:opacity-50 transition-colors"
                          title="Save"
                        >
                          <Check size={14} />
                        </button>
                      </div>
                    </div>
                  ) : (
                    <p className="text-gray-800 break-words leading-relaxed pl-0.5">{ann.comment}</p>
                  )}
                </div>
              ))}
            </div>
          )}
        </section>

      </div>

      {/* Footer / Actions */}
      <div className="p-5 border-t border-gray-200 bg-white flex flex-col gap-3">
        <div className="space-y-1">
          <label className="block text-xs font-semibold text-gray-700">Designer Name</label>
          <input 
            type="text" 
            required
            disabled={isExporting}
            className="w-full px-2.5 py-1.5 border border-gray-300 rounded focus:ring-1 focus:ring-brand-500 focus:border-brand-500 text-xs outline-none"
            placeholder="e.g. John Doe"
            value={designerName}
            onChange={(e) => setDesignerName(e.target.value)}
          />
        </div>
        
        <div className="space-y-1">
          <label className="block text-xs font-semibold text-gray-700">Job Number</label>
          <input 
            type="text" 
            required
            disabled={isExporting}
            className="w-full px-2.5 py-1.5 border border-gray-300 rounded focus:ring-1 focus:ring-brand-500 focus:border-brand-500 text-xs outline-none"
            placeholder="e.g. JB-2023-04"
            value={jobNumber}
            onChange={(e) => setJobNumber(e.target.value)}
          />
        </div>

        <div className="space-y-1">
          <label className="block text-xs font-semibold text-gray-700">Delivery Type</label>
          <select 
            disabled={isExporting}
            className="w-full px-2.5 py-1.5 border border-gray-300 rounded focus:ring-1 focus:ring-brand-500 focus:border-brand-500 text-xs outline-none bg-white cursor-pointer"
            value={deliveryType}
            onChange={(e) => setDeliveryType(e.target.value as any)}
          >
            <option value="Low">Low</option>
            <option value="Medium">Medium</option>
            <option value="Complex">Complex</option>
          </select>
        </div>

        <button
          onClick={() => onDownload({ designerName, jobNumber, deliveryType })}
          disabled={!file || !isFormValid || isExporting}
          className="w-full flex items-center justify-center gap-2 px-4 py-2 bg-gray-900 text-white rounded-lg text-sm font-medium hover:bg-gray-800 focus:outline-none focus:ring-2 focus:ring-offset-2 focus:ring-gray-900 disabled:opacity-50 disabled:cursor-not-allowed transition-all mt-2"
        >
          {isExporting ? (
            <>
              <svg className="animate-spin h-4 w-4 text-white" xmlns="http://www.w3.org/2000/svg" fill="none" viewBox="0 0 24 24">
                <circle className="opacity-25" cx="12" cy="12" r="10" stroke="currentColor" strokeWidth="3"></circle>
                <path className="opacity-75" fill="currentColor" d="M4 12a8 8 0 018-8V0C5.373 0 0 5.373 0 12h4zm2 5.291A7.962 7.962 0 014 12H0c0 3.042 1.135 5.824 3 7.938l3-2.647z"></path>
              </svg>
              <span>Exporting...</span>
            </>
          ) : (
            <>
              <Download size={16} />
              <span>Download PDF & CSV</span>
            </>
          )}
        </button>
      </div>
    </div>
  );
};
